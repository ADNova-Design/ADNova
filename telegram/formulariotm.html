document.getElementById('myForm').addEventListener('submit', function(event) { var isValid = true; var inputs = this.elements; for (var i = 0; i < inputs.length; i++) { if (inputs[i].type === "text" || inputs[i].type === "email") { if (!inputs[i].value) { isValid = false; alert('Por favor completa todos los campos'); break; } } } if (!isValid) { event.preventDefault(); } });
